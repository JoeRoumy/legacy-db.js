/**
 * @file websocket.js
 * @fileOverview Factory module for the websocket back-end class.
 * @author Andreas Olofsson (andreas@erisindustries.com)
 * @module rpc/websocket
 */

'use strict';

var logger = require('winston');
var WebSocket = require('ws');
var backEnd = require('./back_end');
var rpc = require('./rpc');

/**
 * Create a new instance of the WebSocketBackEnd class.
 * @param {string} URL - The endpoint URL
 * @returns {WebSocketBackEnd}
 */
exports.createInstance = function(URL){
    return new WebSocketBackEnd(URL);
};

/**
 * The WebSocketBackEnd class does RPCs over websockets.
 *
 * @param {string} URL - The endpoint URL.
 * @augments module:rpc/backend~NetworkingBackEnd
 * @constructor
 */
function WebSocketBackEnd(URL){
    backEnd.NetworkingBackEnd.call(URL);
    /**
     * Storing requests until their responses comes back.
     *
     * @type TODO
     * @private
     */
    this._reqs = {};

    /**
     * The current request id. It is incremented by 1 for each request that's made.
     *
     * @type {number}
     * @private
     */
    this._currentId = 1;

    /**
     * The websocket client object.
     * @private
     */
    this._ws = null;

}

/**
 * Start the websocket service.
 *
 * @param {module:rpc/backend~netStartCallback} callback - Called when connected to the RPC server.
 * @override
 */
WebSocketBackEnd.prototype.start = function(callback){

    this._ws = new WebSocket(this._URL);

    (function(wsep){

        // When the connection is established.
        wsep._ws.on('open', function open() {
            logger.info("Websocket connection established.");
            wsep._isRunning = true;
            callback();
        });

        // When a message arrives.
        wsep._ws.on('message', function(data, flags) {
            var resp;

            try {
                resp = JSON.parse(data);
            } catch (err) {
                logger.error("Error when parsing websocket response: " + err);
                return;
            }
            if (!rpc.isResponse(resp)) {
                logger.error("Response object is not a proper json-rpc 2.0 response object.", response);
                return;
            }
            var cFunc = wsep._reqs[resp.id];
            if (!cFunc) {
                logger.error("Response object does not have a callback.", response);
            }

            delete wsep._reqs[resp.id];

            if (!resp.error){
                logger.debug(typeof cFunc);
                cFunc(null, resp.result);
            } else {
                cFunc(resp.error);
            }
        });

        // When the connection is shut down.
        // TODO some teardown.
        wsep._ws.on('close', function close() {
            logger.info("Websocket connection closed.");
        });

    })(this);
};

/**
 * Send a message. This method is used for all RPC (RMI) methods.
 *
 * @param {string} method - The method name.
 * @param {?Object} params - The parameters object.
 * @param {module:rpc/rpc~methodCallback} callback - The callback function.
 * @override
 */
WebSocketBackEnd.prototype.send = function(method, params, callback){

    if(!this._isRunning){
        throw new Error("The websocket service has not been started.");
    }

    logger.debug("Callback: ", typeof callback);

    var id = this._nextId();

    var req = rpc.request(id, method, params);

    var reqs = this._reqs;

    this._ws.send(JSON.stringify(req), function ack(error) {
        if (error) {
            logger.error("Websocket error: ", error);
            callback(error);
        } else {
            reqs[id] = callback;
        }
    });

};

/**
 * TODO
 *
 * @param {netShutdownCallback} callback - Called when everything has been shut down.
 * @override
 */
WebSocketBackEnd.prototype.shutDown = function(callback){
    this._isRunning = false;
    callback();
};


/**
 * Get the next id.
 *
 * @returns {string} id - The id as a string.
 * @private
 */
WebSocketBackEnd.prototype._nextId = function(){
    return (this._currentId++).toString();
};
