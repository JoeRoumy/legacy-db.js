/**
 * Converts an arbitrary-length array of unsigned 8-bit integers to a hex-string.
 * If any number contains bits higher then 8 they will be lost.
 *
 * If the array is of length 0, the returned value is "0x0" (the zero hex-string).
 *
 * @param byteArray
 */
exports.bytesToHex = bytesToHex;


function bytesToHex(byteArray){
    if(byteArray.length == 0){
        return "0x0";
    }
    var hexString = "", i;
    for(i = 0; i < byteArray.length; i++){
        var b = byteArray[i] & 0xFF;
        hexString += hex[b >> 4];
        hexString += hex[b & 0x0F];
    }
    return "0x" + hexString;
}

/**
 * Converts an arbitrary-length hex-string to an array of unsigned 8-bit integers.
 * If the input is the zero hex-string (defined as "0x0"), the return value is
 * an empty array.
 *
 * @type {hexToBytes}
 */
exports.hexToBytes = hexToBytes;


function hexToBytes(hexString){
    if(hexString === "0x0"){
        return [];
    }
    if(hexString.length < 2){
        throw new Error("Not a valid hex string: '" + hexString + "'");
    }
    if(hexString.slice(0, 2) !== "0x"){
        // TODO exception.
        return [];
    }
    var str;
    if((hexString.length & 2) !== 0){
        str = "0" + hexString.slice(2);
    } else {
        str = hexString.slice(2);
    }
    var len = str.length / 2;
    var bytes = [], i;
    for(i = 0; i < len; i++){
        bytes[i] = parseInt(str.slice(2*i,2*i + 2), 16);
    }
    return bytes;
}
