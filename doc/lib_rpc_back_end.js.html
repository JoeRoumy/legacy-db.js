/**

 * @module rpc/backend
 */

exports.NetworkingBackEnd = NetworkingBackEnd;

/**
 * <tt>NetworkingBackEnd</tt> is the base class for all networking back-ends used by the library.
 * @constructor
 */
function NetworkingBackEnd(URL) {
    if(!URL){
        throw new Error("No URL provided.");
    }

    this._URL = URL;
    this._isRunning = false;
}

/**
 * @param {netStartCallback} callback - Called when everything is set up.
 */
NetworkingBackEnd.prototype.start = function(callback){
    this._isRunning = true;
    callback();
};

/**
 *
 * @param {netShutdownCallback} callback - Called when everything has been shut down.
 */
NetworkingBackEnd.prototype.shutDown = function(callback){
    this._isRunning = false;
    callback();
};

/**
 * Is the system running?
 *
 * @returns {boolean}
 */
NetworkingBackEnd.prototype.isRunning = function(){
    return this._isRunning;
};
